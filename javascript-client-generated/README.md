```javascript

      parser: {
        amd: false
```

## Getting Started

Please follow the [installation](#installation) instruction and execute the following JS code:

```javascript
var PatientsApi = require('patients_api');

var api = new PatientsApi.DefaultApi()
var body = new PatientsApi.Patient(); // {Patient} Patient to add

var callback = function(error, data, response) {
  if (error) {
    console.error(error);
  } else {
    console.log('API called successfully.');
  }
};
api.addPatient(body, callback);
```
